    print("2. חיסור")

    if בחירה == '1':
        תוצאה = num1 + num2
        print(f"{num1} + {num2} = {תוצאה}")
    elif בחירה == '2':
        תוצאה = num1 - num2
        print(f"{num1} - {num2} = {תוצאה}")
    elif בחירה == '3':
        תוצאה = num1 * num2
        print(f"{num1} × {num2} = {תוצאה}")
    elif בחירה == '4':
        if num2 != 0:
            תוצאה = num1 / num2
            print(f"{num1} ÷ {num2} = {תוצאה}")
        else:
            print("❌ לא ניתן לחלק באפס!")
    else:
        print("❌ בחירה לא תקינה!")

import random

def צ_אט_בוט():
    responses = {
        "שלום": ["שלום! איך אפשר לעזור?", "היי! מה שלומך?", "ברוך הבא!"],
        "איך קוראים לך": ["אני הבוט של Python", "קוראים לי PyBot", "אני עוזר הווירטואלי שלך"],
        "מה זה python": ["Python זה שפת תכנות מדהימה!", "השפה הכי פופולרית בעולם!", "שפה פשוטה וחזקה"],
        "תודה": ["אין בעד מה!", "בכיף לעזור!", "תמיד פה בשבילך"],
        "להתראות": ["להתראות! היה כיף!", "בוי בוי!", "נתראה בפעם הבאה!"]
    }
    
    print("🤖 שלום! אני הצ'אט בוט של Python")
    print("כתבו 'יציאה' כדי לסיים")
    
    while True:
        user_input = input("\n👤 אתם: ").lower().strip()
        
        if user_input == 'יציאה':
            print("🤖 בוט: להתראות!")
            break
            
        response_found = False
        for key, responses_list in responses.items():
            if key in user_input:
                response = random.choice(responses_list)
                print(f"🤖 בוט: {response}")
                response_found = True
                break
        
        if not response_found:
            print("🤖 בוט: מעניין... ספר לי יותר!")

אוטומציה - בדיקת אתרים

import requests
import time
from datetime import datetime

def בדיקת_אתרים():
    """בדיקה אם אתרים פועלים"""
    אתרים = [
        "https://www.google.com",
        "https://www.github.com", 
        "https://www.python.org",
        "https://www.stackoverflow.com"
    ]
    
    print("🌐 בודק סטטוס אתרים...")
    print("-" * 40)
    
    for אתר in אתרים:
        try:
            start_time = time.time()
            response = requests.get(אתר, timeout=5)
            response_time = round((time.time() - start_time) * 1000, 2)
            
            if response.status_code == 200:
                status = "✅ פועל"
                color = "\033[92m"  # ירוק
            else:
                status = f"⚠️ שגיאה {response.status_code}"
                color = "\033[93m"  # צהוב
                
            print(f"{color}{אתר:<30} {status} ({response_time}ms)\033[0m")
            
        except Exception as e:
            print(f"\033[91m{אתר:<30} ❌ לא זמין\033[0m")
    
    print(f"\n⏰ נבדק ב: {datetime.now().strftime('%H:%M:%S')}")

# הרצה
בדיקת_אתרים()

ניתוח נתונים - סטטיסטיקות על קובץ טקסט

import string
from collections import Counter

def ניתוח_טקסט(טקסט):
    """ניתוח סטטיסטי של טקסט"""
    
    # סינון טקסט
    טקסט_נקי = טקסט.lower()
    טקסט_ללא_פיסוק = טקסט_נקי.translate(str.maketrans('', '', string.punctuation))
    
    # חישובים בסיסיים
    מספר_תווים = len(טקסט)
    מספר_תווים_ללא_רווחים = len(טקסט.replace(' ', ''))
    מספר_מילים = len(טקסט_ללא_פיסוק.split())
    מספר_משפטים = טקסט.count('.') + טקסט.count('!') + טקסט.count('?')
    
    # מילים נפוצות
    מילים = טקסט_ללא_פיסוק.split()
    מילים_נפוצות = Counter(מילים).most_common(5)
    
    # הדפסת תוצאות
    print("📊 סטטיסטיקות הטקסט:")
    print("=" * 30)
    print(f"📝 תווים: {מספר_תווים:,}")
    print(f"✏️  תווים ללא רווחים: {מספר_תווים_ללא_רווחים:,}")
    print(f"📖 מילים: {מספר_מילים:,}")
    print(f"📋 משפטים: {מספר_משפטים}")
    print(f"📏 אורך ממוצע של מילה: {round(מספר_תווים_ללא_רווחים/מספר_מילים, 1)} תווים")
    
    print("\n🔥 המילים הנפוצות ביותר:")
    for i, (מילה, כמות) in enumerate(מילים_נפוצות, 1):
        print(f"{i}. '{מילה}' - {כמות} פעמים")

# דוגמה
טקסט_לדוגמה = """
Python היא שפת תכנות מדהימה. היא פשוטה וחזקה.
Python מאפשרת לפתח במהירות. לכן Python כל כך פופולרית.
רבים אוהבים את Python כי היא ברורה וקלה ללמידה.
"""
